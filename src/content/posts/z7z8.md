### 素数判断
```c++
#include <bits/stdc++.h>
using namespace std;
bool isPrime(int n)
{
    if(n < 2) return false;
    for(int i = 2; i * i <= n; i++){
        if(n % i == 0) return 0;
    }
    return true;
}
signed main()
{
    int a;
    cin >> a;
    if(isPrime(a)){
        cout << "是素数";
    } 
    else{
        cout << "不是素数";
    }
    return 0;
}
```
